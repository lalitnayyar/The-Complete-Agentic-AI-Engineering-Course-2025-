def calculate_series(n):
    total = 0.0
    for i in range(n):
        term = (-1)**i / (2 * i + 1)
        total += term
    return total * 4

n_terms = 10000
result = calculate_series(n_terms)
print(f"Approximation of pi using {n_terms} terms: {result}")
